** Line numbers
#+begin_src emacs-lisp
  (global-display-line-numbers-mode 1)

  (setq-default
   display-line-numbers-width 5
   display-line-numbers-type 'relative)

  ; Disable line numbers for some modes
  (dolist
  	 (mode '(term-mode-hook
              shell-mode-hook
              eshell-mode-hook))
  	 (add-hook mode (lambda () (display-line-numbers-mode 0))))
#+end_src
